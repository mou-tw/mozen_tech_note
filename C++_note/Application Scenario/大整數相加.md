實際應用中，可能遇到超大位數的整數相加，可能超過long能表達的上限
對應的方式就是將每個字符拆開，並遍歷相加
```
#include<iostream>
using namespace std;

int main(){
    string n1 = "789465132489874512345454915";
    string n2 = "7894651324894915";

    string ret;
    int carry =0;
    
    //num pointer
    int p1 = n1.size() -1;
    int p2 = n2.size() -1;

    while (p1 >=0 || p2 >=0)
    {
        int x = (p1 >= 0) ? (n1[p1] - '0') : 0;
        int y = (p2 >= 0) ? (n2[p2] - '0') : 0;
        int sum = x+y+carry;

        carry = sum / 10;
        ret += (sum% 10 + '0');
        --p1;
        --p2;
    }

    int i = 0, j = ret.size()-1;
    
    while (i<j) {
        char tmp = ret[i];
        ret[i] = ret[j];
        ret[j] = tmp;
        ++i;
        --j;
    }
    cout << n1 << "+" << n2 << "=" << ret<< endl;
}
```
